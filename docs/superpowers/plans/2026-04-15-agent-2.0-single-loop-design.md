# Agent 2.0 Single-Loop Tool-Calling Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在不回归 1.0 功能的前提下，落地可测试的单 Agent 闭环（Planner/Executor + 原生 tool-calling + progress 事件 + 自愈 + 上下文压缩 + 单任务锁 + 幻觉率评测）。

**Architecture:** 主链路放在 `src/main/services/agent/`，由 planner-executor 负责模型调用、工具执行、终止控制与 trace。AI provider 扩展原生 tool-calling 协议，IPC 增加 `ai:agentRun` / `ai:agentAbort` 和实时事件通道，前端在 ChatPanel 增加 Agent 模式入口并渲染进度。测试分为 provider/tools/planner/lock 四层，评测由 `evals/agent-v2` 输出完成率、工具正确率、平均步数与幻觉率。

**Tech Stack:** Electron, React, Zustand, TypeScript, OpenAI-compatible Chat Completions, Vitest, tsx

```ts
async chatWithTools(messages: AIChatMessage[], tools: AIToolDefinition[], options?: { temperature?: number; maxTokens?: number; toolChoice?: 'auto' | 'none' }): Promise<AIChatWithToolsResult> {
  const { headers, body } = this.buildRequestBody(messages, options, false, tools, options?.toolChoice ?? 'auto')
  const response = await fetch(this.config.baseURL, { method: 'POST', headers, body })
  if (!response.ok) throw new Error(`AI API error: ${await response.text()}`)

  const data = await response.json()
  const choice = data.choices?.[0]
  const message = choice?.message ?? {}
  const toolCalls = Array.isArray(message.tool_calls)
    ? message.tool_calls.filter((item: any) => item?.type === 'function' && item?.function?.name)
    : []

  return {
    content: typeof message.content === 'string' ? message.content : '',
    toolCalls,
    finishReason: typeof choice?.finish_reason === 'string' ? choice.finish_reason : null
  }
}
```

```ts
export const TOOL_NAMES = ['extract_job_page', 'get_profile', 'save_battle_card'] as const
export type AgentToolName = (typeof TOOL_NAMES)[number]

export function createAgentTools(deps: AgentToolDeps) {
  async function execute(input: { name: string; argumentsJson: string }): Promise<AgentToolExecutionResult> {
    if (!TOOL_NAMES.includes(input.name as AgentToolName)) {
      return { ok: false, errorCode: 'TOOL_NOT_ALLOWED', message: `Tool not allowed: ${input.name}` }
    }

    let parsed: Record<string, unknown> = {}
    try {
      parsed = input.argumentsJson ? JSON.parse(input.argumentsJson) : {}
    } catch {
      return { ok: false, errorCode: 'TOOL_ARGUMENT_INVALID', message: 'arguments must be valid JSON' }
    }

    if (input.name === 'extract_job_page') {
      const webContentId = Number(parsed.webContentId)
      if (!Number.isFinite(webContentId)) {
        return { ok: false, errorCode: 'TOOL_ARGUMENT_INVALID', message: 'webContentId is required' }
      }
      return { ok: true, data: await deps.extractJobPage(webContentId) }
    }

    if (input.name === 'get_profile') {
      const profile = deps.getProfile()
      return {
        ok: true,
        data: {
          resumeTextSnippet: (profile.resumeText || '').trim().slice(0, 1200),
          selfIntroTextSnippet: (profile.selfIntroText || '').trim().slice(0, 600),
          hasResume: Boolean((profile.resumeText || '').trim())
        }
      }
    }

    const payload = parsed as SaveBattleCardToolInput
    if (!payload.companyName || !payload.positionName || !payload.sourceUrl) {
      return { ok: false, errorCode: 'TOOL_ARGUMENT_INVALID', message: 'companyName/positionName/sourceUrl are required' }
    }

    return { ok: true, data: await deps.saveBattleCard(payload) }
  }

  return { execute }
}
```

```ts
export async function runPlannerExecutor(input: RunPlannerExecutorInput): Promise<AgentRunResult> {
  const maxSteps = input.maxSteps ?? 6
  const messages: AIChatMessage[] = [
    { role: 'system', content: AGENT_SYSTEM_PROMPT },
    { role: 'user', content: buildInitialUserPrompt(input.request) }
  ]

  const trace: AgentTrace = { runId: input.request.runId, steps: 0, toolCalls: [], errors: [] }
  let repeatedSameCallCount = 0
  let previousCallFingerprint = ''

  for (let step = 1; step <= maxSteps; step += 1) {
    trace.steps = step
    input.onProgress?.({ runId: input.request.runId, step, phase: 'model', message: `step ${step}: planning` })

    const response = await input.provider.chatWithTools(messages, input.toolSchemas, { toolChoice: 'auto' })

    if (!response.toolCalls.length) {
      return {
        finalAnswer: response.content || '任务结束，但模型未返回文本答案。',
        trace,
        artifacts: extractArtifacts(trace)
      }
    }

    for (const call of response.toolCalls) {
      const fingerprint = `${call.function.name}:${call.function.arguments}`
      repeatedSameCallCount = fingerprint === previousCallFingerprint ? repeatedSameCallCount + 1 : 0
      previousCallFingerprint = fingerprint

      if (repeatedSameCallCount >= 2) {
        const recoverMessage = '工具调用参数重复失败，请调整参数或向用户解释阻塞点。'
        messages.push({ role: 'system', content: recoverMessage })
      }
      if (repeatedSameCallCount >= 3) {
        return {
          finalAnswer: '检测到重复工具调用死循环，任务已提前中止。',
          trace: {
            ...trace,
            errors: [...trace.errors, 'CIRCUIT_BREAKER: repeated tool call >= 3']
          },
          artifacts: extractArtifacts(trace)
        }
      }

      input.onProgress?.({ runId: input.request.runId, step, phase: 'tool', message: `calling ${call.function.name}` })
      const startedAt = Date.now()
      const toolResult = await input.tools.execute({ name: call.function.name, argumentsJson: call.function.arguments })
      const durationMs = Date.now() - startedAt

      trace.toolCalls.push({
        step,
        toolName: call.function.name as any,
        argumentsJson: call.function.arguments,
        ok: toolResult.ok,
        durationMs,
        errorCode: toolResult.ok ? undefined : toolResult.errorCode
      })

      messages.push({ role: 'assistant', content: response.content || '', toolCalls: [call] } as any)
      messages.push({
        role: 'tool',
        toolCallId: call.id,
        toolName: call.function.name,
        content: JSON.stringify(compressToolPayload(step, call.function.name, toolResult))
      } as any)

      if (!toolResult.ok) {
        messages.push({
          role: 'system',
          content: '上一个工具调用失败。请先分析错误原因，再决定重试、改参或向用户说明阻塞点；禁止无限重复相同参数。'
        })
      }
    }
  }

  return {
    finalAnswer: '达到最大步骤上限，任务已中止。',
    trace,
    artifacts: extractArtifacts(trace)
  }
}
```

```ts
function compressToolPayload(
  step: number,
  toolName: string,
  toolResult: AgentToolExecutionResult
): Record<string, unknown> {
  if (!toolResult.ok) {
    return {
      step,
      toolName,
      ok: false,
      errorCode: toolResult.errorCode,
      message: toolResult.message
    }
  }

  // 第一步可保留较多字段，后续轮次只保留结构化摘要与关键事实
  if (toolName === 'extract_job_page') {
    const raw = toolResult.data as {
      title?: string
      url?: string
      content?: string
      salaryRange?: string
      requirementsSummary?: string
      companyName?: string
      positionName?: string
    }
    return {
      step,
      toolName,
      ok: true,
      compact: true,
      title: raw.title || '',
      url: raw.url || '',
      companyName: raw.companyName || '',
      positionName: raw.positionName || '',
      salaryRange: raw.salaryRange || '',
      requirementsSummary: raw.requirementsSummary || '',
      contentSnippet: step === 1 ? String(raw.content || '').slice(0, 1200) : undefined,
      contentDropped: step > 1
    }
  }

  return {
    step,
    toolName,
    ok: true,
    data: toolResult.data
  }
}
```

```ts
export function createSingleRunLock() {
  let activeRunId: string | null = null
  return {
    tryAcquire(runId: string) {
      if (activeRunId) return false
      activeRunId = runId
      return true
    },
    release(runId: string) {
      if (activeRunId === runId) activeRunId = null
    },
    isBusy() {
      return activeRunId !== null
    }
  }
}
```

```ts
ipcMain.on('ai:agentRun', async (event, payload: AgentRunRequest, requestId: string) => {
  const runId = `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`

  if (!runLock.tryAcquire(runId)) {
    sendStreamEvent(event.sender, 'ai:agentRun:error', requestId, JSON.stringify({ code: 'AGENT_BUSY', message: 'Agent is already running' }))
    return
  }

  try {
    const result = await runAgentWorkflow({
      ...payload,
      runId,
      onProgress: (progress) => sendStreamEvent(event.sender, 'ai:agentRun:progress', requestId, JSON.stringify(progress))
    })

    sendStreamEvent(event.sender, 'ai:agentRun:done', requestId, JSON.stringify(result))
  } catch (error: any) {
    sendStreamEvent(event.sender, 'ai:agentRun:error', requestId, JSON.stringify({ code: 'AGENT_FAILED', message: error?.message || 'unknown error' }))
  } finally {
    runLock.release(runId)
  }
})
```

```ts
function calcHallucinationRate(records: EvalRecord[]): number {
  const checked = records.filter((item) => item.savedCard && item.facts)
  if (!checked.length) return 0

  let mismatch = 0
  for (const item of checked) {
    const card = item.savedCard!
    const facts = item.facts!
    const companyMismatch = normalize(card.companyName) !== normalize(facts.companyName)
    const positionMismatch = normalize(card.positionName) !== normalize(facts.positionName)
    const salaryMismatch = normalize(card.salaryRange || '') !== normalize(facts.salary || '')
    if (companyMismatch || positionMismatch || salaryMismatch) mismatch += 1
  }

  return mismatch / checked.length
}
```
